# ghostfs Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Zbudować własny, niekompatybilny z istniejącymi system plików w C, trzymający dane w autorskim pliku-kontenerze, montowany przez FUSE i zarządzany przez CLI.

**Architecture:** Blokowy układ on-disk z i-węzłami (styl ext2/xv6). Cała logika w bibliotece `libghostfs`; FUSE i CLI to cienkie nakładki na to samo API rdzenia. Budowa warstwami od dołu: `block → alloc → super → inode → dir → fs → cli → fuse`.

**Tech Stack:** C (C11), libfuse3, GNU Make, własny mini-harness testowy (`tests/test.h`), AddressSanitizer.

## Global Constraints

- Język: **C11** (`-std=c11`), kompilacja z `-Wall -Wextra -Werror`.
- Rozmiar bloku: **4096 B** (`GHOST_BLOCK_SIZE`).
- Magic superbloku: bajty `"GHOSTFS\1"` = `{0x47,0x48,0x4F,0x53,0x54,0x46,0x53,0x01}`.
- I-węzeł on-disk: **256 B** → 16 i-węzłów na blok.
- Wskaźniki na blok: `512` (`4096/8`); `NDIRECT = 12`.
- Wszystkie testy uruchamiane także pod **AddressSanitizer** (`make test-asan`) — zero wycieków/błędów pamięci dopuszczalne.
- FUSE: **libfuse3** (`pkg-config --cflags --libs fuse3`).
- Liczby wielobajtowe zapisywane w **little-endian** kolejności hosta (zakładamy x86-64; brak konwersji w v1 — świadome YAGNI).
- Każde zadanie kończy się zielonymi testami i commitem.

```c
#define GHOSTFS_H
#include <stdint.h>

#define GH_BLOCK_SIZE   4096u
#define GH_MAGIC        "GHOSTFS\x01"   /* 8 bajtów */
#define GH_MAGIC_LEN    8
#define GH_NDIRECT      12
#define GH_PTRS_PER_BLK (GH_BLOCK_SIZE / 8)   /* 512 */
#define GH_INODE_SIZE   256u
#define GH_INODES_PER_BLK (GH_BLOCK_SIZE / GH_INODE_SIZE) /* 16 */
#define GH_NAME_MAX     255
#define GH_ROOT_INO     1

enum gh_itype { GH_FREE = 0, GH_FILE = 1, GH_DIR = 2 };

struct gh_superblock {            /* leży w bloku 0, dopełniony do 4096 B */
    uint8_t  magic[GH_MAGIC_LEN];
    uint32_t block_size;
    uint64_t total_blocks;
    uint64_t inode_count;
    uint64_t bitmap_start;
    uint64_t inode_start;
    uint64_t data_start;
    uint64_t root_inode;
};

struct gh_inode {                 /* dokładnie GH_INODE_SIZE bajtów */
    uint16_t type;
    uint16_t mode;
    uint32_t uid;
    uint32_t gid;
    uint32_t nlink;
    uint64_t size;
    uint64_t atime, mtime, ctime;
    uint64_t direct[GH_NDIRECT];
    uint64_t indirect;
    uint64_t double_indirect;
    uint8_t  pad[GH_INODE_SIZE - (2+2+4+4+4+8+8+8+8 + 8*GH_NDIRECT + 8 + 8)];
};

struct gh_dirent {                /* rekord stałej długości w pliku katalogu */
    uint64_t ino;                 /* 0 = wpis wolny/usunięty */
```

```c
#include "block.h"
#include "ghostfs.h"
#include <fcntl.h>
#include <unistd.h>
#include <errno.h>
#include <string.h>

int gh_dev_create(const char *path, uint64_t total_blocks, struct gh_dev *dev) {
    int fd = open(path, O_RDWR | O_CREAT | O_TRUNC, 0600);
    if (fd < 0) return -errno;
    off_t bytes = (off_t)total_blocks * GH_BLOCK_SIZE;
    if (ftruncate(fd, bytes) != 0) { int e = -errno; close(fd); return e; }
    dev->fd = fd; dev->total_blocks = total_blocks;
    return 0;
}

int gh_dev_open(const char *path, struct gh_dev *dev) {
    int fd = open(path, O_RDWR);
    if (fd < 0) return -errno;
    off_t end = lseek(fd, 0, SEEK_END);
    if (end < 0) { int e = -errno; close(fd); return e; }
    dev->fd = fd;
    dev->total_blocks = (uint64_t)end / GH_BLOCK_SIZE;
    return 0;
}

void gh_dev_close(struct gh_dev *dev) {
    if (dev && dev->fd >= 0) { close(dev->fd); dev->fd = -1; }
}

static int io_at(struct gh_dev *dev, uint64_t blkno, void *buf, int write) {
    if (blkno >= dev->total_blocks) return -EINVAL;
    off_t off = (off_t)blkno * GH_BLOCK_SIZE;
    ssize_t n = write ? pwrite(dev->fd, buf, GH_BLOCK_SIZE, off)
                      : pread(dev->fd, buf, GH_BLOCK_SIZE, off);
    if (n < 0) return -errno;
    if (n != (ssize_t)GH_BLOCK_SIZE) return -EIO;
    return 0;
}

int gh_block_read(struct gh_dev *dev, uint64_t blkno, void *buf) {
    return io_at(dev, blkno, buf, 0);
}
int gh_block_write(struct gh_dev *dev, uint64_t blkno, const void *buf) {
    return io_at(dev, blkno, (void *)buf, 1);
}
```

- [ ] **Step 4: Zaimplementuj `src/alloc.c`**

Mapa bitowa rozciąga się od `bitmap_start` przez tyle bloków, ile trzeba na `total_blocks` bitów. Bit `i` odpowiada blokowi `i` całego kontenera (bity bloków metadanych nigdy nie są zwalniane). Alokujemy tylko bloki `>= data_start`.

```c
#include "alloc.h"
#include <string.h>

static int bitmap_rw(struct gh_dev *dev, const struct gh_superblock *sb,
                     uint64_t blkno, int set_op, int do_write, int *out_set) {
    uint64_t bit = blkno;
    uint64_t byte = bit / 8;
    uint64_t blk_in_map = sb->bitmap_start + byte / GH_BLOCK_SIZE;
    uint64_t off = byte % GH_BLOCK_SIZE;
    uint8_t buf[GH_BLOCK_SIZE];
    int r = gh_block_read(dev, blk_in_map, buf);
    if (r) return r;
    uint8_t mask = (uint8_t)(1u << (bit % 8));
    if (out_set) *out_set = (buf[off] & mask) ? 1 : 0;
    if (do_write) {
        if (set_op) buf[off] |= mask; else buf[off] &= (uint8_t)~mask;
        return gh_block_write(dev, blk_in_map, buf);
    }
    return 0;
}

int gh_bitmap_test(struct gh_dev *dev, const struct gh_superblock *sb,
                   uint64_t blkno, int *set) {
    return bitmap_rw(dev, sb, blkno, 0, 0, set);
}

int gh_free_block(struct gh_dev *dev, const struct gh_superblock *sb, uint64_t blkno) {
    if (blkno < sb->data_start || blkno >= sb->total_blocks) return -EINVAL;
    return bitmap_rw(dev, sb, blkno, 0, 1, NULL);
}

int gh_alloc_block(struct gh_dev *dev, const struct gh_superblock *sb, uint64_t *out) {
    for (uint64_t b = sb->data_start; b < sb->total_blocks; b++) {
        int set = 0;
        int r = gh_bitmap_test(dev, sb, b, &set);
        if (r) return r;
        if (!set) {
            r = bitmap_rw(dev, sb, b, 1, 1, NULL);
            if (r) return r;
            uint8_t zero[GH_BLOCK_SIZE]; memset(zero, 0, sizeof(zero));
            r = gh_block_write(dev, b, zero);   /* czyszczenie nowego bloku */
            if (r) return r;
            *out = b; return 0;
        }
    }
    return -ENOSPC;
}
```

```c
#include "super.h"
#include <string.h>

static uint64_t ceil_div(uint64_t a, uint64_t b) { return (a + b - 1) / b; }

int gh_format(const char *path, uint64_t total_blocks, uint64_t inode_count) {
    if (total_blocks < 8) return -EINVAL;
    /* układ: [0]=superblok, mapa, i-węzły, dane */
    uint64_t bitmap_blocks = ceil_div(total_blocks, GH_BLOCK_SIZE * 8);
    uint64_t inode_blocks  = ceil_div(inode_count, GH_INODES_PER_BLK);
    uint64_t bitmap_start = 1;
    uint64_t inode_start  = bitmap_start + bitmap_blocks;
    uint64_t data_start   = inode_start + inode_blocks;
    if (data_start >= total_blocks) return -EINVAL;

    struct gh_dev dev;
    int r = gh_dev_create(path, total_blocks, &dev);
    if (r) return r;

    struct gh_superblock sb;
    memset(&sb, 0, sizeof(sb));
    memcpy(sb.magic, GH_MAGIC, GH_MAGIC_LEN);
    sb.block_size = GH_BLOCK_SIZE;
    sb.total_blocks = total_blocks;
    sb.inode_count = inode_blocks * GH_INODES_PER_BLK;
    sb.bitmap_start = bitmap_start;
    sb.inode_start = inode_start;
    sb.data_start = data_start;
    sb.root_inode = GH_ROOT_INO;

    /* zapis superbloku (dopełniony do pełnego bloku) */
    uint8_t blk[GH_BLOCK_SIZE]; memset(blk, 0, sizeof(blk));
    memcpy(blk, &sb, sizeof(sb));
    if ((r = gh_block_write(&dev, 0, blk))) { gh_dev_close(&dev); return r; }

    /* wyzeruj mapę i tablicę i-węzłów */
    memset(blk, 0, sizeof(blk));
    for (uint64_t b = bitmap_start; b < data_start; b++)
        if ((r = gh_block_write(&dev, b, blk))) { gh_dev_close(&dev); return r; }

    /* zaznacz w mapie bity bloków metadanych [0, data_start) */
    for (uint64_t b = 0; b < data_start; b++) {
        uint64_t byte = b / 8, off = byte % GH_BLOCK_SIZE;
        uint64_t mblk = bitmap_start + byte / GH_BLOCK_SIZE;
        uint8_t mbuf[GH_BLOCK_SIZE];
        if ((r = gh_block_read(&dev, mblk, mbuf))) { gh_dev_close(&dev); return r; }
        mbuf[off] |= (uint8_t)(1u << (b % 8));
        if ((r = gh_block_write(&dev, mblk, mbuf))) { gh_dev_close(&dev); return r; }
    }

    /* root: i-węzeł GH_ROOT_INO typu katalog, rozmiar 0 (zawartość w Task 5) */
    uint64_t iblk = inode_start + (GH_ROOT_INO / GH_INODES_PER_BLK);
    uint64_t idx  = GH_ROOT_INO % GH_INODES_PER_BLK;
    uint8_t ibuf[GH_BLOCK_SIZE];
    if ((r = gh_block_read(&dev, iblk, ibuf))) { gh_dev_close(&dev); return r; }
    struct gh_inode root; memset(&root, 0, sizeof(root));
    root.type = GH_DIR; root.mode = 0755; root.nlink = 2;
    memcpy(ibuf + idx * GH_INODE_SIZE, &root, sizeof(root));
    if ((r = gh_block_write(&dev, iblk, ibuf))) { gh_dev_close(&dev); return r; }

    gh_dev_close(&dev);
    return 0;
}

int gh_mount_sb(struct gh_dev *dev, struct gh_superblock *sb) {
    uint8_t blk[GH_BLOCK_SIZE];
    int r = gh_block_read(dev, 0, blk);
    if (r) return r;
    if (memcmp(blk, GH_MAGIC, GH_MAGIC_LEN) != 0) return -EINVAL;
    memcpy(sb, blk, sizeof(*sb));
    if (sb->block_size != GH_BLOCK_SIZE) return -EINVAL;
    if (sb->data_start >= sb->total_blocks) return -EINVAL;
    return 0;
}
```

```c
#include <string.h>
#include <time.h>
#include <errno.h>

static void inode_loc(const struct gh_superblock *sb, uint64_t ino,
                      uint64_t *blk, uint64_t *idx) {
    *blk = sb->inode_start + ino / GH_INODES_PER_BLK;
    *idx = ino % GH_INODES_PER_BLK;
}

int gh_inode_read(struct gh_dev *dev, const struct gh_superblock *sb,
                  uint64_t ino, struct gh_inode *out) {
    if (ino >= sb->inode_count) return -EINVAL;
    uint64_t blk, idx; inode_loc(sb, ino, &blk, &idx);
    uint8_t buf[GH_BLOCK_SIZE];
    int r = gh_block_read(dev, blk, buf); if (r) return r;
    memcpy(out, buf + idx * GH_INODE_SIZE, sizeof(*out));
    return 0;
}

int gh_inode_write(struct gh_dev *dev, const struct gh_superblock *sb,
                   uint64_t ino, const struct gh_inode *in) {
    if (ino >= sb->inode_count) return -EINVAL;
    uint64_t blk, idx; inode_loc(sb, ino, &blk, &idx);
    uint8_t buf[GH_BLOCK_SIZE];
    int r = gh_block_read(dev, blk, buf); if (r) return r;
    memcpy(buf + idx * GH_INODE_SIZE, in, sizeof(*in));
    return gh_block_write(dev, blk, buf);
}

int gh_inode_alloc(struct gh_dev *dev, const struct gh_superblock *sb,
                   uint16_t type, uint64_t *out_ino) {
    for (uint64_t ino = GH_ROOT_INO + 1; ino < sb->inode_count; ino++) {
        struct gh_inode n; int r = gh_inode_read(dev, sb, ino, &n);
        if (r) return r;
        if (n.type == GH_FREE) {
            memset(&n, 0, sizeof(n));
            n.type = type; n.mode = (type == GH_DIR) ? 0755 : 0644;
            n.nlink = 1; n.atime = n.mtime = n.ctime = (uint64_t)time(NULL);
            r = gh_inode_write(dev, sb, ino, &n); if (r) return r;
            *out_ino = ino; return 0;
        }
    }
    return -ENOSPC;
}

/* mapuje logiczny numer bloku pliku na fizyczny; alokuje gdy alloc!=0 */
static int bmap(struct gh_dev *dev, const struct gh_superblock *sb,
                struct gh_inode *node, uint64_t lbn, int alloc, uint64_t *out) {
    if (lbn < GH_NDIRECT) {
        if (node->direct[lbn] == 0) {
            if (!alloc) { *out = 0; return 0; }
            int r = gh_alloc_block(dev, sb, &node->direct[lbn]); if (r) return r;
        }
        *out = node->direct[lbn]; return 0;
    }
    lbn -= GH_NDIRECT;
    if (lbn < GH_PTRS_PER_BLK) {
        if (node->indirect == 0) {
            if (!alloc) { *out = 0; return 0; }
            int r = gh_alloc_block(dev, sb, &node->indirect); if (r) return r;
        }
        uint64_t ptrs[GH_PTRS_PER_BLK];
        int r = gh_block_read(dev, node->indirect, ptrs); if (r) return r;
        if (ptrs[lbn] == 0) {
            if (!alloc) { *out = 0; return 0; }
            r = gh_alloc_block(dev, sb, &ptrs[lbn]); if (r) return r;
            r = gh_block_write(dev, node->indirect, ptrs); if (r) return r;
        }
        *out = ptrs[lbn]; return 0;
    }
    lbn -= GH_PTRS_PER_BLK;
    /* double indirect */
    if (lbn < (uint64_t)GH_PTRS_PER_BLK * GH_PTRS_PER_BLK) {
        if (node->double_indirect == 0) {
            if (!alloc) { *out = 0; return 0; }
            int r = gh_alloc_block(dev, sb, &node->double_indirect); if (r) return r;
        }
        uint64_t l1[GH_PTRS_PER_BLK];
        int r = gh_block_read(dev, node->double_indirect, l1); if (r) return r;
        uint64_t i1 = lbn / GH_PTRS_PER_BLK, i2 = lbn % GH_PTRS_PER_BLK;
        if (l1[i1] == 0) {
            if (!alloc) { *out = 0; return 0; }
            r = gh_alloc_block(dev, sb, &l1[i1]); if (r) return r;
            r = gh_block_write(dev, node->double_indirect, l1); if (r) return r;
        }
        uint64_t l2[GH_PTRS_PER_BLK];
        r = gh_block_read(dev, l1[i1], l2); if (r) return r;
        if (l2[i2] == 0) {
            if (!alloc) { *out = 0; return 0; }
            r = gh_alloc_block(dev, sb, &l2[i2]); if (r) return r;
            r = gh_block_write(dev, l1[i1], l2); if (r) return r;
        }
        *out = l2[i2]; return 0;
    }
    return -EFBIG;
}

ssize_t gh_inode_pread(struct gh_dev *dev, const struct gh_superblock *sb,
                       struct gh_inode *node, void *buf, size_t n, uint64_t off) {
    if (off >= node->size) return 0;
    if (off + n > node->size) n = node->size - off;
    size_t done = 0; uint8_t blk[GH_BLOCK_SIZE];
    while (done < n) {
        uint64_t lbn = (off + done) / GH_BLOCK_SIZE;
        uint64_t boff = (off + done) % GH_BLOCK_SIZE;
        size_t chunk = GH_BLOCK_SIZE - boff;
        if (chunk > n - done) chunk = n - done;
        uint64_t phys; int r = bmap(dev, sb, node, lbn, 0, &phys);
        if (r) return r;
        if (phys == 0) memset(blk, 0, GH_BLOCK_SIZE);     /* dziura = zera */
        else { r = gh_block_read(dev, phys, blk); if (r) return r; }
        memcpy((uint8_t*)buf + done, blk + boff, chunk);
        done += chunk;
    }
    return (ssize_t)done;
}

ssize_t gh_inode_pwrite(struct gh_dev *dev, const struct gh_superblock *sb,
                        uint64_t ino, struct gh_inode *node,
                        const void *buf, size_t n, uint64_t off) {
    size_t done = 0; uint8_t blk[GH_BLOCK_SIZE];
    while (done < n) {
        uint64_t lbn = (off + done) / GH_BLOCK_SIZE;
        uint64_t boff = (off + done) % GH_BLOCK_SIZE;
        size_t chunk = GH_BLOCK_SIZE - boff;
        if (chunk > n - done) chunk = n - done;
        uint64_t phys; int r = bmap(dev, sb, node, lbn, 1, &phys);
        if (r) return r;
        if (chunk != GH_BLOCK_SIZE) {              /* czesciowy blok: read-modify-write */
            r = gh_block_read(dev, phys, blk); if (r) return r;
        }
        memcpy(blk + boff, (const uint8_t*)buf + done, chunk);
        r = gh_block_write(dev, phys, blk); if (r) return r;
        done += chunk;
    }
    if (off + n > node->size) node->size = off + n;
    node->mtime = (uint64_t)time(NULL);
    int r = gh_inode_write(dev, sb, ino, node); if (r) return r;
    return (ssize_t)done;
}

int gh_inode_free(struct gh_dev *dev, const struct gh_superblock *sb, uint64_t ino) {
    struct gh_inode n; int r = gh_inode_read(dev, sb, ino, &n); if (r) return r;
    uint64_t nblocks = (n.size + GH_BLOCK_SIZE - 1) / GH_BLOCK_SIZE;
    for (uint64_t lbn = 0; lbn < nblocks; lbn++) {
        uint64_t phys; if (bmap(dev, sb, &n, lbn, 0, &phys) == 0 && phys)
            gh_free_block(dev, sb, phys);
    }
    if (n.indirect) gh_free_block(dev, sb, n.indirect);
    if (n.double_indirect) {
        uint64_t l1[GH_PTRS_PER_BLK];
        if (gh_block_read(dev, n.double_indirect, l1) == 0)
            for (uint64_t i = 0; i < GH_PTRS_PER_BLK; i++)
                if (l1[i]) gh_free_block(dev, sb, l1[i]);
        gh_free_block(dev, sb, n.double_indirect);
    }
    memset(&n, 0, sizeof(n)); n.type = GH_FREE;
    return gh_inode_write(dev, sb, ino, &n);
}
```

```c
int gh_dir_add(struct gh_dev *dev, const struct gh_superblock *sb,

        if (de.ino != 0 && strncmp(de.name, name, GH_NAME_MAX) == 0

        if (de.ino == 0 && !found_free) { free_off = off; found_free = 1; }

        != (ssize_t)DENT_SZ) return -EIO;
    return 0;

    if (r) return r;

        if (de.ino != 0 && de.name_len == strlen(name)

            de.ino = 0;   /* oznacz wolny */
            gh_inode_pwrite(dev, sb, dir_ino, &dir, &de, DENT_SZ, off);

        struct gh_dirent de;

        if (de.ino != 0 && strcmp(de.name, ".") && strcmp(de.name, "..")) {

    if (r) return r;

        struct gh_dirent de;

        if (de.ino != 0) { int c = cb(&de, ctx); if (c) return c; }

    const char *slash = strrchr(path, '/');
    if (!slash) return -EINVAL;
    size_t plen = (size_t)(slash - path);
    if (plen == 0) { strcpy(parent, "/"); }
    else { memcpy(parent, path, plen); parent[plen] = '\0'; }
    if (strlen(slash + 1) > GH_NAME_MAX) return -ENAMETOOLONG;
    strcpy(name, slash + 1);
    return 0;
}

int gh_path_resolve(struct gh_dev *dev, const struct gh_superblock *sb,
                    const char *path, uint64_t *out_ino) {
    if (path[0] != '/') return -EINVAL;
    uint64_t cur = sb->root_inode;
    char comp[GH_NAME_MAX + 1]; size_t ci = 0;
    for (const char *p = path + 1; ; p++) {
        if (*p == '/' || *p == '\0') {
            if (ci > 0) {
                comp[ci] = '\0';
                uint64_t next; int r = gh_dir_lookup(dev, sb, cur, comp, &next);
                if (r) return r;
                cur = next; ci = 0;
            }
            if (*p == '\0') break;
        } else {
            if (ci >= GH_NAME_MAX) return -ENAMETOOLONG;
            comp[ci++] = *p;
        }
    }
    *out_ino = cur; return 0;
}
```

```c
int gh_fs_mount(struct gh_fs *fs, const char *path) {
    int r = gh_dev_open(path, &fs->dev); if (r) return r;
    r = gh_mount_sb(&fs->dev, &fs->sb);
    if (r) { gh_dev_close(&fs->dev); return r; }
    return 0;
}
void gh_fs_unmount(struct gh_fs *fs) { gh_dev_close(&fs->dev); }

int gh_fs_getattr(struct gh_fs *fs, const char *path,
                  struct gh_inode *out, uint64_t *out_ino) {
    uint64_t ino; int r = gh_path_resolve(&fs->dev, &fs->sb, path, &ino);
    if (r) return r;
    r = gh_inode_read(&fs->dev, &fs->sb, ino, out); if (r) return r;
    if (out_ino) *out_ino = ino;
    return 0;
}

static int make_node(struct gh_fs *fs, const char *path, uint16_t type, uint16_t mode) {
    char parent[1024], name[256];
    int r = gh_path_split(path, parent, name); if (r) return r;
    uint64_t pino; r = gh_path_resolve(&fs->dev, &fs->sb, parent, &pino);
    if (r) return r;
    uint64_t exists; if (gh_dir_lookup(&fs->dev, &fs->sb, pino, name, &exists) == 0)
        return -EEXIST;
    uint64_t ino; r = gh_inode_alloc(&fs->dev, &fs->sb, type, &ino); if (r) return r;
    struct gh_inode n; gh_inode_read(&fs->dev, &fs->sb, ino, &n);
    n.mode = mode;
    if (type == GH_DIR) {   /* dopisz . i .. */
        struct gh_dirent de; uint64_t off = 0;
        const char *nm[2] = { ".", ".." }; uint64_t tgt[2] = { ino, pino };
        for (int k = 0; k < 2; k++) {
            memset(&de, 0, sizeof(de)); de.ino = tgt[k];
            de.name_len = (uint16_t)strlen(nm[k]); memcpy(de.name, nm[k], de.name_len);
            gh_inode_pwrite(&fs->dev, &fs->sb, ino, &n, &de, sizeof(de), off);
            off += sizeof(de);
        }
        n.nlink = 2;
    }
    gh_inode_write(&fs->dev, &fs->sb, ino, &n);
    r = gh_dir_add(&fs->dev, &fs->sb, pino, name, ino);
    if (r) { gh_inode_free(&fs->dev, &fs->sb, ino); return r; }
    return 0;
}

int gh_fs_create(struct gh_fs *fs, const char *path, uint16_t mode) {
    return make_node(fs, path, GH_FILE, mode);
}
int gh_fs_mkdir(struct gh_fs *fs, const char *path, uint16_t mode) {
    return make_node(fs, path, GH_DIR, mode);
}

static int remove_node(struct gh_fs *fs, const char *path, int want_dir) {
    uint64_t ino; int r = gh_path_resolve(&fs->dev, &fs->sb, path, &ino);
    if (r) return r;
    if (ino == GH_ROOT_INO) return -EBUSY;
    struct gh_inode n; gh_inode_read(&fs->dev, &fs->sb, ino, &n);
    if (want_dir) {
        if (n.type != GH_DIR) return -ENOTDIR;
        int empty; gh_dir_is_empty(&fs->dev, &fs->sb, ino, &empty);
        if (!empty) return -ENOTEMPTY;
    } else {
        if (n.type == GH_DIR) return -EISDIR;
    }
    char parent[1024], name[256];
    r = gh_path_split(path, parent, name); if (r) return r;
    uint64_t pino; r = gh_path_resolve(&fs->dev, &fs->sb, parent, &pino);
    if (r) return r;
    r = gh_dir_remove(&fs->dev, &fs->sb, pino, name); if (r) return r;
    return gh_inode_free(&fs->dev, &fs->sb, ino);
}

int gh_fs_rmdir(struct gh_fs *fs, const char *path)  { return remove_node(fs, path, 1); }

ssize_t gh_fs_read(struct gh_fs *fs, const char *path, void *buf, size_t n, uint64_t off) {
    uint64_t ino; int r = gh_path_resolve(&fs->dev, &fs->sb, path, &ino);
    if (r) return r;
    struct gh_inode node; gh_inode_read(&fs->dev, &fs->sb, ino, &node);
    if (node.type != GH_FILE) return -EISDIR;
    return gh_inode_pread(&fs->dev, &fs->sb, &node, buf, n, off);

}

int gh_fs_readdir(struct gh_fs *fs, const char *path, gh_dir_iter_fn cb, void *ctx) {
    uint64_t ino; int r = gh_path_resolve(&fs->dev, &fs->sb, path, &ino);
    if (r) return r;
    struct gh_inode node; gh_inode_read(&fs->dev, &fs->sb, ino, &node);
    if (node.type != GH_DIR) return -ENOTDIR;

/* fsck: zbuduj oczekiwaną mapę z osiągalnych i-węzłów, porównaj z zapisaną */
static void mark(uint8_t *want, uint64_t b) { want[b/8] |= (uint8_t)(1u<<(b%8)); }

int gh_fsck(struct gh_fs *fs, int repair, int *issues) {
    struct gh_superblock *sb = &fs->sb;
    size_t mapbytes = (sb->total_blocks + 7) / 8;
    uint8_t *want = calloc(mapbytes, 1);
    if (!want) return -ENOMEM;
    for (uint64_t b = 0; b < sb->data_start; b++) mark(want, b);  /* metadane */

    for (uint64_t ino = 0; ino < sb->inode_count; ino++) {
        struct gh_inode n;
        if (gh_inode_read(&fs->dev, sb, ino, &n) || n.type == GH_FREE) continue;
        uint64_t nblk = (n.size + GH_BLOCK_SIZE - 1) / GH_BLOCK_SIZE;
        for (uint64_t lbn = 0; lbn < nblk; lbn++) {
            uint64_t phys = 0;
            struct gh_inode tmp = n;
            /* reużyj bmap przez pread-niezależną ścieżkę: bezpośrednie */
            if (lbn < GH_NDIRECT) phys = tmp.direct[lbn];
            else {
                uint64_t k = lbn - GH_NDIRECT;
                if (k < GH_PTRS_PER_BLK && tmp.indirect) {
                    uint64_t p[GH_PTRS_PER_BLK];
                    if (gh_block_read(&fs->dev, tmp.indirect, p) == 0) phys = p[k];
                }
            }
            if (phys) mark(want, phys);
        }
        if (n.indirect) mark(want, n.indirect);
        if (n.double_indirect) mark(want, n.double_indirect);
    }

    int diff = 0;
    for (uint64_t b = 0; b < sb->total_blocks; b++) {
        int set = 0; gh_bitmap_test(&fs->dev, sb, b, &set);
        int exp = (want[b/8] >> (b%8)) & 1;
        if (set != exp) diff++;   /* tylko zliczamy; naprawa hurtem nizej */
    }
    if (issues) *issues = diff;
    if (repair && diff) {
        /* nadpisz cala mape oczekiwana */
        uint8_t blk[GH_BLOCK_SIZE];
        uint64_t mapblocks = (sb->total_blocks + GH_BLOCK_SIZE*8 - 1)/(GH_BLOCK_SIZE*8);
        for (uint64_t mb = 0; mb < mapblocks; mb++) {
            memset(blk, 0, sizeof(blk));
            size_t copy = mapbytes - mb*GH_BLOCK_SIZE;
            if (copy > GH_BLOCK_SIZE) copy = GH_BLOCK_SIZE;
            memcpy(blk, want + mb*GH_BLOCK_SIZE, copy);
            gh_block_write(&fs->dev, sb->bitmap_start + mb, blk);
        }
    }
    free(want);
    return 0;
}
```

```c
#include "fs.h"
#include "super.h"
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <fcntl.h>
#include <unistd.h>
#include <errno.h>

static int print_entry(const struct gh_dirent *de, void *ctx) {
    (void)ctx; printf("%s\n", de->name); return 0;
}

static int cmd_format(int argc, char **argv) {
    if (argc < 5) { fprintf(stderr, "format <plik> <bloki> <inody>\n"); return 2; }
    int r = gh_format(argv[2], strtoull(argv[3],0,10), strtoull(argv[4],0,10));
    if (r) { fprintf(stderr, "format: %s\n", strerror(-r)); return 1; }
    return 0;
}

static int cmd_ls(int argc, char **argv) {
    if (argc < 4) { fprintf(stderr, "ls <plik> <ścieżka>\n"); return 2; }
    struct gh_fs fs; int r = gh_fs_mount(&fs, argv[2]);
    if (r) { fprintf(stderr, "mount: %s\n", strerror(-r)); return 1; }
    r = gh_fs_readdir(&fs, argv[3], print_entry, NULL);
    gh_fs_unmount(&fs);
    if (r) { fprintf(stderr, "ls: %s\n", strerror(-r)); return 1; }
    return 0;
}

static int cmd_put(int argc, char **argv) {
    if (argc < 5) { fprintf(stderr, "put <plik> <src> <ścieżka>\n"); return 2; }
    struct gh_fs fs; if (gh_fs_mount(&fs, argv[2])) return 1;
    int in = open(argv[3], O_RDONLY);
    if (in < 0) { perror("open src"); gh_fs_unmount(&fs); return 1; }
    gh_fs_create(&fs, argv[4], 0644);
    char buf[65536]; ssize_t k; uint64_t off = 0;
    while ((k = read(in, buf, sizeof(buf))) > 0) {
        if (gh_fs_write(&fs, argv[4], buf, (size_t)k, off) != k) {
            fprintf(stderr, "write failed\n"); close(in); gh_fs_unmount(&fs); return 1;
        }
        off += (uint64_t)k;
    }
    close(in); gh_fs_unmount(&fs); return 0;
}

static int cmd_get(int argc, char **argv) {
    if (argc < 5) { fprintf(stderr, "get <plik> <ścieżka> <dst>\n"); return 2; }
    struct gh_fs fs; if (gh_fs_mount(&fs, argv[2])) return 1;
    struct gh_inode st; uint64_t ino;
    if (gh_fs_getattr(&fs, argv[3], &st, &ino)) {
        fprintf(stderr, "brak: %s\n", argv[3]); gh_fs_unmount(&fs); return 1;
    }
    int out = open(argv[4], O_WRONLY|O_CREAT|O_TRUNC, 0644);
    if (out < 0) { perror("open dst"); gh_fs_unmount(&fs); return 1; }
    char buf[65536]; uint64_t off = 0;
    while (off < st.size) {
        ssize_t r = gh_fs_read(&fs, argv[3], buf, sizeof(buf), off);
        if (r <= 0) break;
        if (write(out, buf, (size_t)r) != r) break;
        off += (uint64_t)r;
    }
    close(out); gh_fs_unmount(&fs); return 0;
}

static int cmd_mkdir(int argc, char **argv) {
    if (argc < 4) { fprintf(stderr, "mkdir <plik> <ścieżka>\n"); return 2; }
    struct gh_fs fs; if (gh_fs_mount(&fs, argv[2])) return 1;
    int r = gh_fs_mkdir(&fs, argv[3], 0755); gh_fs_unmount(&fs);
    if (r) { fprintf(stderr, "mkdir: %s\n", strerror(-r)); return 1; }
    return 0;
}

static int cmd_rm(int argc, char **argv) {
    if (argc < 4) { fprintf(stderr, "rm <plik> <ścieżka>\n"); return 2; }
    struct gh_fs fs; if (gh_fs_mount(&fs, argv[2])) return 1;
    int r = gh_fs_unlink(&fs, argv[3]);
    if (r == -EISDIR) r = gh_fs_rmdir(&fs, argv[3]);
    gh_fs_unmount(&fs);
    if (r) { fprintf(stderr, "rm: %s\n", strerror(-r)); return 1; }
    return 0;
}

static int cmd_fsck(int argc, char **argv) {
    if (argc < 3) { fprintf(stderr, "fsck <plik> [--repair]\n"); return 2; }
    int repair = (argc > 3 && strcmp(argv[3], "--repair") == 0);
    struct gh_fs fs; if (gh_fs_mount(&fs, argv[2])) return 1;
    int issues = 0; gh_fsck(&fs, repair, &issues);
    gh_fs_unmount(&fs);
    printf("fsck: %d niespójności%s\n", issues, repair ? " (naprawiono)" : "");
    return issues && !repair ? 1 : 0;
}

int main(int argc, char **argv) {
    if (argc < 2) {
        fprintf(stderr, "uzycie: ghostfs-cli <format|ls|put|get|mkdir|rm|fsck> ...\n");
        return 2;
    }
    const char *c = argv[1];
    if (!strcmp(c, "format")) return cmd_format(argc, argv);
    if (!strcmp(c, "ls"))     return cmd_ls(argc, argv);
    if (!strcmp(c, "put"))    return cmd_put(argc, argv);
    if (!strcmp(c, "get"))    return cmd_get(argc, argv);
    if (!strcmp(c, "mkdir"))  return cmd_mkdir(argc, argv);
    if (!strcmp(c, "rm"))     return cmd_rm(argc, argv);
    if (!strcmp(c, "fsck"))   return cmd_fsck(argc, argv);
    fprintf(stderr, "nieznana komenda: %s\n", c);
    return 2;
}
```

**Interfaces:**

- [ ] **Step 1: Zaimplementuj `src/fuse_main.c`**

```c
#define FUSE_USE_VERSION 31
#include <fuse3/fuse.h>
#include "fs.h"
#include <string.h>
#include <errno.h>
#include <stdlib.h>
#include <stdio.h>
#include <sys/file.h>
#include <fcntl.h>
#include <unistd.h>

static struct gh_fs g_fs;
static int g_lock_fd = -1;

static int gf_getattr(const char *path, struct stat *st, struct fuse_file_info *fi) {
    (void)fi; memset(st, 0, sizeof(*st));
    struct gh_inode n; uint64_t ino;
    int r = gh_fs_getattr(&g_fs, path, &n, &ino);
    if (r) return r;
    st->st_mode = (n.type == GH_DIR ? S_IFDIR : S_IFREG) | n.mode;
    st->st_nlink = n.nlink; st->st_size = (off_t)n.size;
    st->st_atime = n.atime; st->st_mtime = n.mtime; st->st_ctime = n.ctime;
    return 0;
}

struct rd_ctx { void *buf; fuse_fill_dir_t filler; };
static int rd_cb(const struct gh_dirent *de, void *c) {
    struct rd_ctx *x = c;
    x->filler(x->buf, de->name, NULL, 0, 0);
    return 0;
}
static int gf_readdir(const char *path, void *buf, fuse_fill_dir_t filler,
                      off_t off, struct fuse_file_info *fi, enum fuse_readdir_flags fl) {
    (void)off; (void)fi; (void)fl;
    struct rd_ctx x = { buf, filler };
    return gh_fs_readdir(&g_fs, path, rd_cb, &x);
}

static int gf_create(const char *path, mode_t mode, struct fuse_file_info *fi) {
    (void)fi; return gh_fs_create(&g_fs, path, mode & 0777);
}
static int gf_mkdir(const char *path, mode_t mode) {
    return gh_fs_mkdir(&g_fs, path, mode & 0777);
}
static int gf_unlink(const char *path) { return gh_fs_unlink(&g_fs, path); }
static int gf_rmdir(const char *path)  { return gh_fs_rmdir(&g_fs, path); }
static int gf_open(const char *path, struct fuse_file_info *fi) {
    struct gh_inode n; uint64_t ino; (void)fi;
    return gh_fs_getattr(&g_fs, path, &n, &ino);
}
static int gf_read(const char *path, char *buf, size_t n, off_t off,
                   struct fuse_file_info *fi) {
    (void)fi; ssize_t r = gh_fs_read(&g_fs, path, buf, n, (uint64_t)off);
    return (int)r;
}
static int gf_write(const char *path, const char *buf, size_t n, off_t off,
                    struct fuse_file_info *fi) {
    (void)fi; ssize_t r = gh_fs_write(&g_fs, path, buf, n, (uint64_t)off);
    return (int)r;
}
static int gf_truncate(const char *path, off_t size, struct fuse_file_info *fi) {
    (void)path; (void)size; (void)fi; return 0;  /* v1: minimalne; rozszerzenie pozniej */
}
static int gf_utimens(const char *path, const struct timespec tv[2],
                      struct fuse_file_info *fi) {
    (void)path; (void)tv; (void)fi; return 0;
}

static const struct fuse_operations ops = {
    .getattr = gf_getattr, .readdir = gf_readdir, .create = gf_create,
    .mkdir = gf_mkdir, .unlink = gf_unlink, .rmdir = gf_rmdir,
    .open = gf_open, .read = gf_read, .write = gf_write,
    .truncate = gf_truncate, .utimens = gf_utimens,
};

int main(int argc, char **argv) {
    if (argc < 3) {
        fprintf(stderr, "uzycie: ghostfs <kontener> <punkt_montowania> [opcje fuse]\n");
        return 2;
    }
    const char *container = argv[1];
    /* blokada: tylko jeden proces na kontener */
    g_lock_fd = open(container, O_RDWR);
    if (g_lock_fd < 0) { perror("open kontener"); return 1; }
    if (flock(g_lock_fd, LOCK_EX | LOCK_NB) != 0) {
        fprintf(stderr, "kontener juz w uzyciu (flock)\n"); return 1;
    }
    int r = gh_fs_mount(&g_fs, container);
    if (r) { fprintf(stderr, "mount: %s\n", strerror(-r)); return 1; }

    /* przekaz do FUSE argv bez nazwy kontenera (argv[1]) */
    int fargc = argc - 1;
    char **fargv = malloc(sizeof(char*) * (size_t)fargc);
    fargv[0] = argv[0];
    for (int i = 2; i < argc; i++) fargv[i-1] = argv[i];

    int rc = fuse_main(fargc, fargv, &ops, NULL);
    gh_fs_unmount(&g_fs);
    free(fargv);
    flock(g_lock_fd, LOCK_UN); close(g_lock_fd);
    return rc;
}
```
